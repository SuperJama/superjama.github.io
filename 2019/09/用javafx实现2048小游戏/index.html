用javafx实现2048小游戏

九月 1, 2019 发布在 小游戏, 2048

通过javafx来实现简单的2048小游戏，增强动手能力。

目录

  • 2.源码

1.实现原理

其实2048的运行逻辑并不难，可以用矩阵来模拟。
首先，游戏初始化：
0 0 0 2
2 0 0 2
0 0 2 0
0 0 0 0
通过移动，将相同的数字合并，不相同的移向移动的方向（向右移）：
0 0 0 2
0 0 0 4
0 0 0 2
0 0 0 0
上下左右的移动都是同样的。

举个例子，自己体会：
 0 2 4 8
 2 2 4 4
16 8 4 2
 2 4 4 8
向右移：
 0 2 4 8
 0 0 4 8
16 8 4 2
 0 2 8 8

    // 将数据写入文件
    public void writeData() throws IOException{
        File f = new File("data.txt");
        FileOutputStream fop = new FileOutputStream(f);
        OutputStreamWriter writer = new OutputStreamWriter(fop, "UTF-8");
        for(int i = 4; i >= 0 ; i--){
            writer.append(String.valueOf(dataarray[i]));
            writer.append("\n");
        }

        writer.close();
        fop.close();
    }

    // 读取数据
    public void readData() throws IOException{
        File f = new File("data.txt");

        if(f.exists()){
            System.out.println("文件存在！");
        }
        else{
            System.out.println("文件不存在，创建文件！");
            FileOutputStream fop = new FileOutputStream(f);
            OutputStreamWriter writer = new OutputStreamWriter(fop, "UTF-8");
            writer.append("0\n0\n0\n0\n0\n");
            writer.close();
            fop.close();
        }

        FileInputStream fip = new FileInputStream(f);
        InputStreamReader reader = new InputStreamReader(fip, "UTF-8");
        StringBuffer sb = new StringBuffer();
        while (reader.ready()) {
            sb.append((char) reader.read());
            // 转成char加到StringBuffer对象中
        }
        StringBuffer[] str = new StringBuffer[5];
        int k = 0;
        for(int i = 0; i < 5; i++){
            int m = k;
            str[i] = new StringBuffer();
            for(int j = m; j < sb.length(); j++){
                if(sb.charAt(j) == '\n'){
                    k++;
                    break;
                }
                str[i].append(sb.charAt(j));
                k++;
            }
        }
        for(int i = 0; i < 5; i++){
            dataarray[i] = Integer.parseInt(str[i].toString());
        }
        reader.close();
        fip.close();
    }

    // 获得最高分
    public int getMaxScore(){
        int maxscore = data[0].getScore();
        for(int i = 1; i < 5; i++){
            if(maxscore <= data[i].getScore()){
                maxscore = data[i].getScore();
            }
        }
        return maxscore;
    }

    // 获得最低分
    public int getMinScore(){
        int minscore = data[0].getScore();
        for(int i = 1; i < 5; i++){
            if(minscore >= data[i].getScore()){
                minscore = data[i].getScore();
            }
        }
        return minscore;
    }

    // 获得最低分位置
    public int getMinScorePos(){
        int minscore = data[0].getScore();
        int pos = 0;
        for(int i = 1; i < 5; i++){
            if(minscore >= data[i].getScore()){
                minscore = data[i].getScore();
                pos = i;
            }
        }
        return pos;
    }

    // 游戏结束
    public void gameOver(){
        if(gamePane.compScore() >= getMinScore()){
            dataarray[getMinScorePos()] = gamePane.compScore();
            sortdata();
            try{
                writeData();
            }
            catch (IOException i){
                System.out.println("写入数据错误！");
            }
        }

        Label text = new Label("游戏结束!");
        text.setFont(Font.font("DFKai-SB", 14));
        GridPane gameoverpane = new GridPane();
        gameoverpane.add(text, 0, 0);
        gameoverpane.setAlignment(Pos.CENTER);
        Scene gameover = new Scene(gameoverpane,100, 100);
        Stage gameoverstage = new Stage();
        gameoverstage.setScene(gameover);
        gameoverstage.show();
    }

        }
        public void setName(String fname){
            this.name.set(fname);
        }
        public int getScore(){
            return this.score.get();
        }
        public void setScore(int fscore){
            this.score.set(fscore);
        }
    }

// 主游戏区类
class GamePane extends GridPane{

    /*游戏区所需要的变量*/
    // 画
    Rectangle[][] rectarray = new Rectangle[4][4];
    // 游戏数组
    RectObject[][] gamearray = new RectObject[4][4];
    RectObject[][] last = new RectObject[4][4];
    // 得分
    private int score = 0;


    public GamePane(){


        setPadding(new Insets(10, 10, 10, 10));
        setVgap(2);
        setHgap(2);
        setLayoutX(5);
        setLayoutY(5);


        // 游戏开始界面
        init();
        createGameArray();

    }

    // 初始化
    public void init(){
        // 全部置0
        for(int i = 0; i < 4; i++){
            for(int j = 0; j < 4; j++){
                gamearray[i][j] = new RectObject();
                last[i][j] = new RectObject();
            }
        }
        // 初始化两个2
        int[][] randompos = {{0, 0},{0, 0}};
        {
            int k = 0;
            while(k != 2){
                for(int j = 0; j < 2; j++){
                    randompos[k][j] = (int)(Math.random() * 4);
                }
                k++;
                if(randompos[0][0] == randompos[1][0] && randompos[0][1] == randompos[1][1]){
                    k--;
                }
            }
        }
        gamearray[randompos[0][0]][randompos[0][1]].setValue(2);
        gamearray[randompos[1][0]][randompos[1][1]].setValue(2);
    }

    // 是否加一个2
    public boolean isAddTwo(){
        if(chackArray()){
            return true;
        }
        return false;
    }

    // 随机添加一个2或4
    public void addTwo(){
        int num_i = (int)(Math.random() * 4);
        int num_j = (int)(Math.random() * 4);
        boolean flag = true;
        while(flag){
            if(gamearray[num_i][num_j].getValue() == 0){
                if(Math.random() > 0.6){
                    gamearray[num_i][num_j].setValue(2);
                }
                else{
                    gamearray[num_i][num_j].setValue(4);
                }
                flag = false;
            }
            else{
                num_i = (int)(Math.random() * 4);
                num_j = (int)(Math.random() * 4);
            }
        }
    }

    // 画出游戏数组
    public void createGameArray(){
        for(int i = 0; i < 4; i++){
            for(int j = 0; j < 4; j++){
                rectarray[i][j] = new Rectangle(50, 50);
                rectarray[i][j].setArcWidth(12);
                rectarray[i][j].setArcHeight(12);
                switch(gamearray[i][j].getValue()){
                    case 0: rectarray[i][j].setFill(Color.rgb(188,214,221)); break;
                    case 2: rectarray[i][j].setFill(Color.rgb(173,217,216)); break;
                    case 4: rectarray[i][j].setFill(Color.rgb(159,220,212)); break;
                    case 8: rectarray[i][j].setFill(Color.rgb(145,224,208)); break;
                    case 16: rectarray[i][j].setFill(Color.rgb(131,227,204)); break;
                    case 32: rectarray[i][j].setFill(Color.rgb(117,231,200)); break;
                    case 64: rectarray[i][j].setFill(Color.rgb(103,234,196)); break;
                    case 128: rectarray[i][j].setFill(Color.rgb(88,237,192)); break;
                    case 256: rectarray[i][j].setFill(Color.rgb(74,241,188)); break;
                    case 512: rectarray[i][j].setFill(Color.rgb(60,244,184)); break;
                    case 1024: rectarray[i][j].setFill(Color.rgb(46,248,180)); break;
                    case 2048: rectarray[i][j].setFill(Color.rgb(32,251,176)); break;
                    default: rectarray[i][j].setFill(Color.rgb(18,255,172));
                }
                add(rectarray[i][j], j, i);
                if(gamearray[i][j].getValue() != 0){
                    Label a = new Label(String.valueOf(gamearray[i][j].getValue()));
                    a.setFont(Font.font("Mangal", FontWeight.BOLD, 16));
                    a.setAlignment(Pos.CENTER);
                    a.setMinSize(50, 50);
                    add(a, j, i);
                }
            }
        }
    }

    // 复制原数组
    public void copyArray(){
        for(int i = 0; i < 4; i++){
            for(int j = 0; j < 4; j++){
                last[i][j].setValue(gamearray[i][j].getValue());
            }
        }
    }

    // 检查数组是否改变
    public boolean chackArray(){
        for(int i = 0; i < 4; i++){
            for(int j = 0; j < 4; j++){
                if(last[i][j].getValue() != gamearray[i][j].getValue())
                    return true;
            }
        }
        return false;
    }

    // 左移
    public void moveLeft(){
        // 移动
        for(int i = 0; i < 4; i++){
            for(int j = 0; j < 3; j++){
                if(gamearray[i][j].getValue() == 0){
                    gamearray[i][j].setValue(gamearray[i][j + 1].getValue());
                    gamearray[i][j + 1].setValue(0);
                }
            }
        }
        for(int i = 0; i < 4; i++){
            for(int j = 0; j < 2; j++){
                if(gamearray[i][j].getValue() == 0){
                    gamearray[i][j].setValue(gamearray[i][j + 1].getValue());
                    gamearray[i][j + 1].setValue(0);
                }
            }
        }
        for(int i = 0; i < 4; i++){
            if(gamearray[i][0].getValue() == 0){
                gamearray[i][0].setValue(gamearray[i][1].getValue());
                gamearray[i][1].setValue(0);
            }
        }
        // 相加
        for(int i = 0; i < 4; i++){
            if(gamearray[i][0].getValue() == gamearray[i][1].getValue()){
                gamearray[i][0].setValue(gamearray[i][0].getValue() * 2);
                gamearray[i][1].setValue(0);
                score += gamearray[i][0].getValue();
                if(gamearray[i][2].getValue() == gamearray[i][3].getValue()){
                    gamearray[i][2].setValue(gamearray[i][2].getValue() * 2);
                    gamearray[i][3].setValue(0);
                    gamearray[i][1].setValue(gamearray[i][2].getValue());
                    gamearray[i][2].setValue(0);
                    score += gamearray[i][2].getValue();
                }
                else{
                    gamearray[i][1].setValue(gamearray[i][2].getValue());
                    gamearray[i][2].setValue(gamearray[i][3].getValue());
                    gamearray[i][3].setValue(0);
                }
            }
            else if(gamearray[i][1].getValue() == gamearray[i][2].getValue()){
                gamearray[i][1].setValue(gamearray[i][1].getValue() * 2);
                gamearray[i][2].setValue(gamearray[i][3].getValue());
                gamearray[i][3].setValue(0);
                score += gamearray[i][1].getValue();
            }
            else if(gamearray[i][2].getValue() == gamearray[i][3].getValue()){
                gamearray[i][2].setValue(gamearray[i][2].getValue() * 2);
                gamearray[i][3].setValue(0);
                score += gamearray[i][2].getValue();
            }
        }
    }

    // 右移

        for(int i = 0; i < 4; i++){
            for(int j = 3; j > 0; j--){
                if(gamearray[i][j].getValue() == 0){
                    gamearray[i][j].setValue(gamearray[i][j - 1].getValue());
                    gamearray[i][j - 1].setValue(0);
                }
            }
        }
        for(int i = 0; i < 4; i++){
            for(int j = 3; j > 1; j--){
                if(gamearray[i][j].getValue() == 0){
                    gamearray[i][j].setValue(gamearray[i][j - 1].getValue());
                    gamearray[i][j - 1].setValue(0);
                }
            }
        }
        for(int i = 0; i < 4; i++){
            if(gamearray[i][3].getValue() == 0){
                gamearray[i][3].setValue(gamearray[i][2].getValue());
                gamearray[i][2].setValue(0);
            }
        }
        // 相加
        for(int i = 0; i < 4; i++){
            if(gamearray[i][3].getValue() == gamearray[i][2].getValue()){
                gamearray[i][3].setValue(gamearray[i][3].getValue() * 2);
                gamearray[i][2].setValue(0);
                score += gamearray[i][3].getValue();
                if(gamearray[i][1].getValue() == gamearray[i][0].getValue()){
                    gamearray[i][1].setValue(gamearray[i][1].getValue() * 2);
                    gamearray[i][0].setValue(0);
                    gamearray[i][2].setValue(gamearray[i][1].getValue());
                    gamearray[i][1].setValue(0);
                    score += gamearray[i][1].getValue();
                }
                else{
                    gamearray[i][2].setValue(gamearray[i][1].getValue());
                    gamearray[i][1].setValue(gamearray[i][0].getValue());
                    gamearray[i][0].setValue(0);
                }
            }
            else if(gamearray[i][2].getValue() == gamearray[i][1].getValue()){
                gamearray[i][2].setValue(gamearray[i][2].getValue() * 2);
                gamearray[i][1].setValue(gamearray[i][0].getValue());

                score += gamearray[i][1].getValue();
            }
        }
    }

    // 上移
    public void moveUp(){
        // 移动
        for(int j = 0; j < 4; j++){
            for(int i = 0; i < 3; i++){
                if(gamearray[i][j].getValue() == 0){
                    gamearray[i][j].setValue(gamearray[i + 1][j].getValue());
                    gamearray[i + 1][j].setValue(0);
                }
            }
        }
        for(int j = 0; j < 4; j++){
            for(int i = 0; i < 2; i++){
                if(gamearray[i][j].getValue() == 0){
                    gamearray[i][j].setValue(gamearray[i + 1][j].getValue());
                    gamearray[i + 1][j].setValue(0);
                }
            }
        }
        for(int i = 0; i < 4; i++){
            if(gamearray[0][i].getValue() == 0){
                gamearray[0][i].setValue(gamearray[1][i].getValue());
                gamearray[1][i].setValue(0);
            }
        }
        // 相加
        for(int i = 0; i < 4; i++){
            if(gamearray[0][i].getValue() == gamearray[1][i].getValue()){
                gamearray[0][i].setValue(gamearray[0][i].getValue() * 2);
                gamearray[1][i].setValue(0);
                score += gamearray[0][i].getValue();
                if(gamearray[2][i].getValue() == gamearray[3][i].getValue()){
                    gamearray[2][i].setValue(gamearray[2][i].getValue() * 2);
                    gamearray[3][i].setValue(0);
                    gamearray[3][i].setValue(gamearray[2][i].getValue());
                    gamearray[2][i].setValue(0);
                    score += gamearray[2][i].getValue();
                }
                else{
                    gamearray[1][i].setValue(gamearray[2][i].getValue());
                    gamearray[2][i].setValue(gamearray[3][i].getValue());
                    gamearray[3][i].setValue(0);
                }
            }
            else if(gamearray[1][i].getValue() == gamearray[2][i].getValue()){
                gamearray[1][i].setValue(gamearray[1][i].getValue() * 2);
                gamearray[2][i].setValue(gamearray[3][i].getValue());
                gamearray[3][i].setValue(0);
                score += gamearray[1][i].getValue();

            else if(gamearray[2][i].getValue() == gamearray[3][i].getValue()){
                gamearray[2][i].setValue(gamearray[2][i].getValue() * 2);
                gamearray[3][i].setValue(0);
                score += gamearray[2][i].getValue();
            }
        }
    }

    // 是否游戏结束
    public boolean isGameOver(){
        for(int i = 0; i < 4; i++){
            for(int j = 0; j < 4; j++){
                if(gamearray[i][j].getValue() == 0){
                    return false;
                }
            }
        }
        for(int i = 0; i < 4; i++){
            for(int j = 0; j < 3; j++){
                if(gamearray[i][j].getValue() == gamearray[i][j + 1].getValue()){
                    return false;
                }
                if(gamearray[j][i].getValue() == gamearray[j + 1][i].getValue()){
                    return false;
                }
            }
        }
        return true;
    }
